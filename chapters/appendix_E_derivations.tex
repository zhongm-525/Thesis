\chapter{Derivations}
\label{app:derivations}

This appendix collects the mathematical derivations whose full steps were quoted in abbreviated form in Chapter~\ref{chap:methods}, in accordance with the handbook guidance that important relationships appear in the main text and their derivations in the appendix.

\section{Single-segment PCC inverse kinematics}
\label{app:deriv_pcc}

The single-segment piecewise-constant-curvature model assumes a soft segment whose backbone traces a circular arc of constant curvature $\kappa = \theta / L$, where $L$ is the arc length and $\theta$ is the total bending angle. The arc lies in a plane that makes an angle $\varphi$ (the azimuth) with a reference direction in the base frame. Without loss of generality the reference direction is the $+x$ axis, and the base tangent is $+z$.

\paragraph{Forward kinematics.}
The arc can be generated by rotating the base frame about the bending-plane normal by a continuous angle $s/r$ as the arc-length parameter $s$ sweeps from $0$ to $L$, with $r = L/\theta$ the radius of curvature. Integrating the resulting tangent yields the tip position
\begin{equation}
  \mathbf{p}_{\mathrm{tip}} = \frac{L}{\theta} \begin{pmatrix} (1 - \cos\theta)\cos\varphi \\ (1 - \cos\theta)\sin\varphi \\ \sin\theta \end{pmatrix}.
  \label{eq:deriv_pcc_fk}
\end{equation}

\paragraph{Inverse from $\mathbf{p}_{\mathrm{tip}} = (x, y, z)$ to $(L, \theta, \varphi)$.}
Let $\rho = \sqrt{x^{2} + y^{2}}$ denote the horizontal projection of the tip. The first two components of Eq.~\eqref{eq:deriv_pcc_fk} give
$x = (L/\theta)(1 - \cos\theta)\cos\varphi$ and $y = (L/\theta)(1 - \cos\theta)\sin\varphi$, so their ratio determines the azimuth,
\begin{equation}
  \varphi = \operatorname{atan2}(y, x).
  \label{eq:deriv_pcc_phi}
\end{equation}
The third component gives $z = (L/\theta)\sin\theta$, and the magnitude of the first two components gives $\rho = (L/\theta)(1 - \cos\theta)$. Their ratio is
$\rho / z = (1 - \cos\theta)/\sin\theta = \tan(\theta/2)$.
Squaring both sides, $\tan^{2}(\theta/2) = \rho^{2}/z^{2}$, and the half-angle identity $\cos\theta = (1 - \tan^{2}(\theta/2))/(1 + \tan^{2}(\theta/2))$ gives
\begin{equation}
  \theta = \arccos\!\left(\frac{z^{2} - (x^{2} + y^{2})}{z^{2} + (x^{2} + y^{2})}\right).
  \label{eq:deriv_pcc_theta}
\end{equation}
Substituting $\theta$ back into $z = (L/\theta)\sin\theta$ yields
\begin{equation}
  L = \frac{z \, \theta}{\sin\theta}.
  \label{eq:deriv_pcc_L}
\end{equation}
Equations~\eqref{eq:deriv_pcc_phi}--\eqref{eq:deriv_pcc_L} are the closed-form inverse used in Section~\ref{sec:pcc_single}.

\paragraph{Configuration to chamber lengths.}
For chamber $i$ at azimuth $\alpha_{i}$ around the central axis and radial offset $d$ from it, the chamber's own backbone is a circular arc concentric with the rod backbone but at a radius $r - d\cos(\alpha_{i} - \varphi)$: positive offsets push the chamber toward the outside of the bend, and negative offsets toward the inside. Multiplying by the bending angle $\theta$ gives
\begin{equation}
  \ell_{i} = \theta \, (r - d\cos(\alpha_{i} - \varphi)) = L - \theta \, d \cos(\alpha_{i} - \varphi),
  \label{eq:deriv_pcc_lengths}
\end{equation}
which reproduces the chamber-length map of Section~\ref{sec:pcc_single}.

\section{Two-segment PCC composition and optimisation-based inverse}
\label{app:deriv_pcc_multi}

\paragraph{Per-segment transform.}
Each segment $s \in \{1, 2\}$ is parameterised by $(\theta_{s}, \varphi_{s}, L_{s})$, with $L_{s} = L_{s}^{(0)} + \Delta L_{s}$ accounting for elongation. Applying Eq.~\eqref{eq:deriv_pcc_fk} to segment $s$ yields a tip displacement
\begin{equation}
  \mathbf{t}_{s} = \frac{L_{s}}{\theta_{s}} \begin{pmatrix} (1 - \cos\theta_{s})\cos\varphi_{s} \\ (1 - \cos\theta_{s})\sin\varphi_{s} \\ \sin\theta_{s} \end{pmatrix}.
\end{equation}
The orientation of the segment's tip frame is a rotation by $\theta_{s}$ about the axis $\mathbf{n}_{s} = (-\sin\varphi_{s},\ \cos\varphi_{s},\ 0)$, which is perpendicular to the bending plane; this rotation matrix is denoted $\mathbf{R}_{s}$.

\paragraph{Composition with inter-segment offset.}
The two modules of the I-Support platform are rotated by $\pi/3$ about the local $z$-axis so the six chambers together cover the full actuation circle (Section~\ref{sec:bg_pneumatic_isupport}). Let $\mathbf{R}_{z}(\pi/3)$ denote this fixed rotation. The second-segment tip displacement expressed in the \emph{first}-segment tip frame is $\mathbf{R}_{z}(\pi/3)\, \mathbf{t}_{2}$, and expressed in the \emph{base} frame is $\mathbf{R}_{1} \mathbf{R}_{z}(\pi/3)\, \mathbf{t}_{2}$. Adding this to the first-segment tip gives the compound forward map
\begin{equation}
  \mathbf{p}_{\mathrm{tip}}(\boldsymbol{\xi}) = \mathbf{t}_{1} + \mathbf{R}_{1} \mathbf{R}_{z}(\pi/3)\, \mathbf{t}_{2},
  \label{eq:deriv_pcc_multi_fk}
\end{equation}
where $\boldsymbol{\xi} = (\theta_{1}, \varphi_{1}, \Delta L_{1}, \theta_{2}, \varphi_{2}, \Delta L_{2})^{\top}$.

\paragraph{Optimisation-based inverse.}
The inverse of Eq.~\eqref{eq:deriv_pcc_multi_fk} has no closed form because $\mathbf{R}_{1}$ couples the two segments nonlinearly. Following Keyvanara et al.\ \cite{keyvanara2023geometric}, the inverse is formulated as the box-constrained least-squares problem
\begin{equation}
  \min_{\boldsymbol{\xi} \in \mathcal{B}} \|\mathbf{p}^{*} - \mathbf{p}_{\mathrm{tip}}(\boldsymbol{\xi})\|_{2}^{2},
  \label{eq:deriv_pcc_multi_ik}
\end{equation}
with $\mathcal{B}$ the Cartesian product of per-parameter bounds specified in Section~\ref{sec:pcc_multi}. Sequential least-squares quadratic programming (SLSQP) solves Eq.~\eqref{eq:deriv_pcc_multi_ik} by constructing, at each iterate $\boldsymbol{\xi}^{(k)}$, a quadratic approximation of the objective and a linearisation of the bounds, and by solving the resulting quadratic programme for a feasible step $\Delta \boldsymbol{\xi}^{(k)}$. The step is accepted if it reduces the objective while remaining inside $\mathcal{B}$; otherwise a line search along $\Delta \boldsymbol{\xi}^{(k)}$ shortens it until a feasible descent is obtained. The SLSQP routine terminates when the objective gradient falls below a tolerance or when the iteration cap is reached. Chapter~\ref{chap:methods} uses the scipy implementation with the bounds and initialisation stated in Section~\ref{sec:pcc_multi}.

\section{Damped least-squares Jacobian update}
\label{app:deriv_dls}

Let $\mathbf{f}: \mathbb{R}^{n_{p}} \to \mathbb{R}^{3}$ denote the trained forward-kinematics network, $\mathbf{x}^{*}$ the target tip, $\mathbf{p}_{k}$ the current pressure estimate, and $\mathbf{x}_{k} = \mathbf{f}(\mathbf{p}_{k})$ and $\mathbf{e}_{k} = \mathbf{x}^{*} - \mathbf{x}_{k}$ the current forward evaluation and residual. Let $\mathbf{J}_{k} = \partial \mathbf{f} / \partial \mathbf{p}|_{\mathbf{p}_{k}}$ be the $3 \times n_{p}$ Jacobian at the iterate.

\paragraph{Newton linearisation.}
A first-order expansion of $\mathbf{f}$ near $\mathbf{p}_{k}$ gives
\begin{equation}
  \mathbf{f}(\mathbf{p}_{k} + \Delta \mathbf{p}) \approx \mathbf{x}_{k} + \mathbf{J}_{k} \Delta \mathbf{p}.
\end{equation}
Setting the right-hand side equal to $\mathbf{x}^{*}$ and rearranging yields the linear system
\begin{equation}
  \mathbf{J}_{k}\, \Delta \mathbf{p} = \mathbf{e}_{k}.
  \label{eq:deriv_dls_linear}
\end{equation}
When $n_{p} > 3$, Eq.~\eqref{eq:deriv_dls_linear} is under-determined and admits a one-parameter family of solutions; when $\mathbf{J}_{k}$ loses rank (at or near kinematic singularities) it admits none.

\paragraph{Tikhonov regularisation.}
The damped least-squares solution replaces the exact solve of Eq.~\eqref{eq:deriv_dls_linear} with the regularised least-squares problem
\begin{equation}
  \Delta \mathbf{p} = \argmin_{\Delta \mathbf{p}} \Big(\, \|\mathbf{J}_{k} \Delta \mathbf{p} - \mathbf{e}_{k}\|_{2}^{2} + \lambda^{2} \|\Delta \mathbf{p}\|_{2}^{2}\, \Big),
  \label{eq:deriv_dls_objective}
\end{equation}
where $\lambda > 0$ is the damping coefficient. Setting the gradient of the objective with respect to $\Delta \mathbf{p}$ to zero,
\begin{equation}
  2 \mathbf{J}_{k}^{\top}(\mathbf{J}_{k} \Delta \mathbf{p} - \mathbf{e}_{k}) + 2 \lambda^{2} \Delta \mathbf{p} = \mathbf{0},
\end{equation}
and collecting terms gives the normal equations
\begin{equation}
  (\mathbf{J}_{k}^{\top} \mathbf{J}_{k} + \lambda^{2} \mathbf{I}_{n_{p}}) \Delta \mathbf{p} = \mathbf{J}_{k}^{\top} \mathbf{e}_{k},
\end{equation}
which yield the primal form of the DLS update
\begin{equation}
  \Delta \mathbf{p} = (\mathbf{J}_{k}^{\top} \mathbf{J}_{k} + \lambda^{2} \mathbf{I}_{n_{p}})^{-1} \mathbf{J}_{k}^{\top} \mathbf{e}_{k}.
  \label{eq:deriv_dls_primal}
\end{equation}

\paragraph{Dual form via the push-through identity.}
When $n_{p} \ge 3$, the primal form requires inverting an $n_{p} \times n_{p}$ matrix. The equivalent dual form
\begin{equation}
  \Delta \mathbf{p} = \mathbf{J}_{k}^{\top} (\mathbf{J}_{k} \mathbf{J}_{k}^{\top} + \lambda^{2} \mathbf{I}_{3})^{-1} \mathbf{e}_{k}
  \label{eq:deriv_dls_dual}
\end{equation}
requires only a $3 \times 3$ inversion. The equivalence follows from the push-through identity: multiplying the difference of Eqs.~\eqref{eq:deriv_dls_primal} and~\eqref{eq:deriv_dls_dual} on the right by $(\mathbf{J}_{k} \mathbf{J}_{k}^{\top} + \lambda^{2} \mathbf{I}_{3})$ yields
\begin{align}
  & (\mathbf{J}_{k}^{\top} \mathbf{J}_{k} + \lambda^{2} \mathbf{I}_{n_{p}})^{-1} \mathbf{J}_{k}^{\top} (\mathbf{J}_{k} \mathbf{J}_{k}^{\top} + \lambda^{2} \mathbf{I}_{3}) \notag \\
  & \quad = (\mathbf{J}_{k}^{\top} \mathbf{J}_{k} + \lambda^{2} \mathbf{I}_{n_{p}})^{-1} (\mathbf{J}_{k}^{\top} \mathbf{J}_{k} + \lambda^{2} \mathbf{I}_{n_{p}}) \mathbf{J}_{k}^{\top} = \mathbf{J}_{k}^{\top},
\end{align}
so the two right-hand sides are equal. Chapter~\ref{chap:methods} uses the dual form of Eq.~\eqref{eq:deriv_dls_dual} because Dataset 1 has $n_{p} = 6$ and Dataset 2 has $n_{p} = 3$; in both cases the $3 \times 3$ solve is at or below the cost of the $n_{p} \times n_{p}$ alternative. The step is then applied with a stabilising scale factor,
\begin{equation}
  \mathbf{p}_{k+1} = \mathbf{p}_{k} + \alpha \, \Delta \mathbf{p},
  \label{eq:deriv_dls_update}
\end{equation}
which completes one Jacobian-learned inverse-kinematics iteration (Eq.~\eqref{eq:dls_update}).
